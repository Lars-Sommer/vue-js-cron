# Types

type Field = {
    id: string,
    items: Array<{
        text: string
        alt: string
        value: string
    }>
}
type Period = {
    id: string,
    text: string,
    value: string[]
}
type Locale = {
    periodPrefix: string,
    periodSuffix: string
    eachPeriod: {
        eachField: {
            empty: string,
            value: string,
            range: string,
            everyX: string,
            prefix: string,
            suffix: string,
        },
        '{{fieldId}}Field': {
            empty: string,
            value: string,
            range: string,
            everyX: string,
            prefix: string,
            suffix: string,
        }
    }
    '{{periodId}}Period': {
        eachField: {
            empty: string,
            value: string,
            range: string,
            everyX: string,
            prefix: string,
            suffix: string,
        },
        '{{fieldId}}Field': {
            empty: string,
            value: string,
            range: string,
            everyX: string,
            prefix: string,
            suffix: string,
        }
    }
}
